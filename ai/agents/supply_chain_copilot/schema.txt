-- ==========================================
-- USERS
-- ==========================================

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    role VARCHAR(50) DEFAULT 'Analyst',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- CHAT SESSIONS
-- ==========================================

CREATE TABLE chat_sessions (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(id) ON DELETE CASCADE,
    session_title VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- CHAT MESSAGES
-- ==========================================

CREATE TABLE chat_messages (
    id SERIAL PRIMARY KEY,
    session_id INT REFERENCES chat_sessions(id) ON DELETE CASCADE,
    sender VARCHAR(20) CHECK(sender IN ('USER','AI')),
    message TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- AI REQUESTS
-- ==========================================

CREATE TABLE ai_requests (
    id SERIAL PRIMARY KEY,
    session_id INT REFERENCES chat_sessions(id),
    user_query TEXT NOT NULL,
    detected_intent VARCHAR(50),
    processing_time FLOAT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- AI RESPONSES
-- ==========================================

CREATE TABLE ai_responses (
    id SERIAL PRIMARY KEY,
    request_id INT REFERENCES ai_requests(id) ON DELETE CASCADE,
    response TEXT NOT NULL,
    confidence DECIMAL(4,2),
    generated_by VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- SCENARIO RUNS
-- ==========================================

CREATE TABLE scenario_runs (
    id SERIAL PRIMARY KEY,
    session_id INT REFERENCES chat_sessions(id),
    scenario_name VARCHAR(150),
    duration_days INT,
    gdp_loss DECIMAL(8,2),
    inflation DECIMAL(6,2),
    fuel_shortage DECIMAL(6,2),
    oil_price DECIMAL(10,2),
    risk_score DECIMAL(5,2),
    executed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- STRATEGIC RECOMMENDATIONS
-- ==========================================

CREATE TABLE strategic_recommendations (
    id SERIAL PRIMARY KEY,
    scenario_id INT REFERENCES scenario_runs(id) ON DELETE CASCADE,
    recommendation_type VARCHAR(50),
    title VARCHAR(255),
    recommendation TEXT,
    confidence DECIMAL(4,2),
    priority VARCHAR(20),
    estimated_impact TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- SQL LOGS
-- ==========================================

CREATE TABLE sql_logs (
    id SERIAL PRIMARY KEY,
    request_id INT REFERENCES ai_requests(id),
    generated_sql TEXT,
    execution_time FLOAT,
    rows_returned INT,
    executed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- AGENT LOGS
-- ==========================================

CREATE TABLE agent_logs (
    id SERIAL PRIMARY KEY,
    request_id INT REFERENCES ai_requests(id),
    agent_name VARCHAR(100),
    status VARCHAR(30),
    execution_time FLOAT,
    output_summary TEXT,
    executed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- RESPONSE SOURCES
-- ==========================================

CREATE TABLE response_sources (
    id SERIAL PRIMARY KEY,
    response_id INT REFERENCES ai_responses(id),
    source_type VARCHAR(50),
    source_name VARCHAR(255),
    source_reference TEXT
);

-- ==========================================
-- USER FEEDBACK
-- ==========================================

CREATE TABLE feedback (
    id SERIAL PRIMARY KEY,
    response_id INT REFERENCES ai_responses(id),
    rating INT CHECK (rating BETWEEN 1 AND 5),
    comments TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- SAVED SIMULATIONS
-- ==========================================

CREATE TABLE saved_simulations (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(id),
    scenario_id INT REFERENCES scenario_runs(id),
    title VARCHAR(255),
    notes TEXT,
    saved_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- CONVERSATION MEMORY
-- ==========================================

CREATE TABLE conversation_memory (
    id SERIAL PRIMARY KEY,
    session_id INT REFERENCES chat_sessions(id),
    memory_key VARCHAR(100),
    memory_value TEXT,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- AI ACTIONS
-- ==========================================

CREATE TABLE ai_actions (
    id SERIAL PRIMARY KEY,
    request_id INT REFERENCES ai_requests(id),
    action_name VARCHAR(100),
    parameters JSONB,
    result JSONB,
    executed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
